### [糖果 ](https://www.acwing.com/problem/content/1171/)

```java


import java.io.IOException;
import java.io.InputStreamReader;
import java.io.StreamTokenizer;
import java.util.*;

public class Main {
    static final int N = (int) (3e5 + 100);
    static int[] head = new int[N];
    static int[] ver = new int[N];
    static int[] nex = new int[N];
    static int[] edge = new int[N];
    static int tot = 0;

    static void addedge(int x, int y, int z) {
        ver[++ tot] = y;
        nex[tot] = head[x];
        head[x] = tot;
        edge[tot] = z;
    }

    static boolean[] in = new boolean[N];
    static int[] dis = new int[N];
    static int[] cnt = new int[N];
    static boolean spfa(int s) {
        Stack<Integer> queue = new Stack<>();
        Arrays.fill(dis, Integer.MIN_VALUE);
        dis[s] = 0;
        cnt[s] = 0;
        queue.push(s);
        while(queue.size() > 0) {
            int x = queue.peek(); queue.pop();
            in[x] = false;
            for(int i = head[x]; i != 0; i = nex[i]) {
                int y = ver[i];
                if(dis[y] < dis[x] + edge[i]) {
                    dis[y] = dis[x] + edge[i];
                    cnt[y] = cnt[x] + 1;
                    if(cnt[y] > n) return true;
                    if(!in[y]) {
                        in[y] = true;
                        queue.add(y);
                    }
                }
            }
        }
        return false;
    }
    static int n, m;
    static StreamTokenizer cin = new StreamTokenizer(new InputStreamReader(System.in));
    static int read() throws IOException {
        cin.nextToken();
        return (int) cin.nval;

    }
    public static void main(String[] args) throws IOException {
        Scanner scanner = new Scanner(System.in);
        n = read();
        m = read();
        for(int i = 1; i <= m; i ++) {
            int z =read();
            int x = read();
            int y = read();
            if(z == 1) {
                addedge(x, y , 0);
                addedge(y, x, 0);
            }
            else if(z == 2) {
                addedge(x, y , 1);
            }
            else if(z == 3) {
                addedge(y, x, 0);
            }
            else if(z == 4) {
                addedge(y, x, 1);
            }
            else if(z == 5) {
                addedge(x, y, 0);
            }
        }

        for(int i = 1; i <= n; i ++) {
            addedge(0, i, 1);
        }

        if(spfa(0)) {
            System.out.println(-1);
            return;
        }
        long ans = 0;
        for(int i = 0; i <= n; i ++) {
            ans += dis[i];
        }
        System.out.println(ans);
    }
}
```
