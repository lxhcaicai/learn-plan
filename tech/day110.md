## 算法学习

### [连接格点](https://www.acwing.com/problem/content/1146/)

```java



import java.util.Arrays;
import java.util.Scanner;

public class Main {

    static final int N = (int) (1E6 + 100);

    static class Edge implements Comparable<Edge>{
        int x, y, z;

        public Edge(int x, int y, int z) {
            this.x = x;
            this.y = y;
            this.z = z;
        }

        @Override
        public int compareTo(Edge o) {
            return z - o.z;
        }
    }
    static Edge[] edges = new Edge[N];

    static int[] fa = new int[N];

    static int find(int x) {
        if(x == fa[x]) {
            return x;
        }
        else {
            fa[x] = find(fa[x]);
            return fa[x];
        }
    }

    static int n, m;
    static int cal(int x, int y) {
        return (x - 1) * n + y;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        n = scanner.nextInt();
        m = scanner.nextInt();
        int tot = 0;
        for(int j = 1; j <= m; j ++) {
            for(int i = 1; i + 1 <= n; i ++) {
                if(j == 1 && j + 1 <= m) {
                    edges[++ tot] = new Edge(cal(i, j), cal(i, j + 1), 2);
                }
                edges[++ tot] = new Edge(cal(i, j), cal(i + 1, j), 1);
            }
        }
        while(scanner.hasNext()) {
            int x1 = scanner.nextInt();
            int y1 = scanner.nextInt();
            int x2 = scanner.nextInt();
            int y2 = scanner.nextInt();
            edges[++ tot] = new Edge(cal(x1, y1), cal(x2, y2), 0);
        }

        for(int i = 1; i <= n * m; i ++) {
            fa[i] = i;
        }

        Arrays.sort(edges, 1, tot + 1);
        int cnt = 0;
        int ans = 0;
        for(int i = 1; i <= tot; i ++) {
            int x = find(edges[i].x);
            int y = find(edges[i].y);
            if(x != y) {
                fa[x] = y;
                ans += edges[i].z;
                cnt ++;
            }
            if(cnt == n*m - 1) break;
        }
        System.out.println(ans);
     }
}
```
