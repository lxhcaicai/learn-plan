### [最大公约数](https://www.acwing.com/problem/content/222/)

```java

import java.util.LinkedList;
import java.util.List;
import java.util.Scanner;

public class Main {
    static final int N = (int) (1e7 + 10);
    static int[] vis = new int[N];
    static int[] phi = new int[N];
    static int cnt = 0;
    static long[] sum = new long[N];
    static int[] primes = new int[N];
    static void getPhi(int n) {
        phi[1] = 0;
        for(int i = 2; i <= n; i ++) {
            if(vis[i] == 0) {
                primes[++ cnt] = i;
                phi[i] = i - 1;
            }
            for(int j = 1; j <= cnt && (long)i * primes[j] <= n; j ++) {
                vis[i * primes[j]] = 1;
                if(i % primes[j] == 0) {
                    phi[i * primes[j]] = phi[i] * primes[j];
                }
                else {
                    phi[i * primes[j]] = phi[i] * phi[primes[j]];
                }
            }
        }
        for(int i = 1; i <= n; i ++) {
            sum[i] = sum[i - 1] + phi[i];
        }
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        getPhi(n);
        long ans = 0;
        for(int i = 1; i <= cnt; i ++) {
            int p = primes[i];
            ans += sum[n / p] * 2 + 1;
        }

        System.out.println(ans);
    }
}
```

### [ Hankson的趣味题](https://www.acwing.com/problem/content/202/)

```java

import java.util.Scanner;

public class Main {
    static final int N = (int) (1e5 + 100);

    static int[] primes = new int[N];
    static int cnt = 0;
    static int[] vis = new int[N];
    static void getprimes(int n) {
        for(int i = 2; i <= n; i ++) {
            if(vis[i] == 0) {
                primes[++ cnt] = i;
            }
            for(int j = 1; j <= cnt && (long)i * primes[j] <= n; j ++) {
                vis[i * primes[j]] = 1;
                if(i % primes[j] == 0) break;
            }
        }
    }

    static int a0, a1, b0,b1;

    static class Factor{
        int num;
        int cnt;

        public Factor(int num, int cnt) {
            this.num = num;
            this.cnt = cnt;
        }
    }
    static int tot = 0;
    static Factor[] factors = new Factor[N];

    static int ans = 0;
    static int gcd(int x, int y) {
        if(y == 0) return x;
        else {
            return gcd(y, x%y);
        }
    }
    static void dfs(int d, int s) {
        if(d > tot) {
            if(gcd(s,a0) == a1 && (long) s * b0/ gcd(s,b0)==b1) ans++;
            return;
        }
        for(int i = 0; i <= factors[d].cnt; i ++) {
            dfs(d + 1, s);
            s *= factors[d].num;
        }
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        getprimes(N - 1);
        int n = scanner.nextInt();
        for(; n > 0; n --) {
            a0 = scanner.nextInt();
            a1 = scanner.nextInt();
            b0 = scanner.nextInt();
            b1 = scanner.nextInt();

            tot = 0;
            int x = b1;
            for(int i = 1; primes[i] * primes[i] <= x; i ++) {
                if(x % primes[i] == 0) {
                    int cnt = 0;
                    while(x % primes[i] == 0) {
                        cnt ++;
                        x /= primes[i];
                    }
                    factors[++ tot]  = new Factor(primes[i],cnt );
                }
            }
            if(x > 1) {
                factors[++ tot] = new Factor(x, 1);
            }
            ans = 0;
            dfs(1, 1);
            System.out.println(ans);
        }
    }
}
```
